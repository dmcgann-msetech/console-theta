Successful Explicit-Path CSS Replacement

Command Used
$path = "D:\mse-console-staging\mse-console\index.html"

$text = [System.IO.File]::ReadAllText(
    $path,
    [System.Text.UTF8Encoding]::new($false)
)

$pattern = '(?s)<style>\s*:root\{.*?</style>'
$replacement = '<link rel="stylesheet" href="css/core.css">'

$newText = [regex]::Replace($text, $pattern, $replacement, 1)

if ($newText -eq $text) {
    throw "CSS block replacement failed. No changes made."
}

[System.IO.File]::WriteAllText(
    $path,
    $newText,
    [System.Text.UTF8Encoding]::new($true)
)
